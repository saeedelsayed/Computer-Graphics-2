file(GLOB_RECURSE SOURCES RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}" "*.cxx")
file(GLOB_RECURSE IMAGES RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}" "*.png")

cgv_add_target(cg_icons
	TYPE plugin NO_EXECUTABLE
	SOURCES ${SOURCES}
	RESOURCES ${IMAGES}
	DEPENDENCIES cgv_base
	OVERRIDE_SHARED_EXPORT_DEFINE ICON_EXPORTS
)

install(TARGETS cg_icons EXPORT cgv_plugins DESTINATION ${CGV_BIN_DEST})
